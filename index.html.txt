    <script>
      const tg = window.Telegram.WebApp;
      tg.expand();

      function sendData() {
        const value = document.getElementById("userInput").value;
        if (!value.trim()) {
          alert("Please enter your name.");
          return;
        }
        tg.sendData(value); // Send data to bot
        tg.close(); // Close the WebView (optional)
      }
    </script>
